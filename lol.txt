local Tool
local a

local OrionLib = loadstring(game:HttpGet(('https://raw.githubusercontent.com/shlexware/Orion/main/source')))()

local Window = OrionLib:MakeWindow({Name = "Blox Fruit", HidePremium = false, SaveConfig = true, ConfigFolder = "OrionTest"})

local Tab = Window:MakeTab({
	Name = "Farm",
	Icon = "rbxassetid://4483345998",
	PremiumOnly = false
})

local Section = Tab:AddSection({
	Name = "Setting"
})

Tab:AddDropdown({
	Name = "Tool",
	Default = "Melee",
	Options = {"Melee", "Sword"},
	Callback = function(Value)
		Tool = Value
	end    
})

local autoBuy = false

Tab:AddToggle({
	Name = "Auto Buy Legendary Aura",
	Default = false,
	Callback = function(Value)
		autoBuy = Value
		if autoBuy then
			a = game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("CommF_"):InvokeServer("ColorsDealer", "1")
			while autoBuy do
				if a == "Winter Sky" or a == "Pure Red" or a == "Snow White" then
					game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("CommF_"):InvokeServer("ColorsDealer", "2")
				end
				wait(1) -- Small delay to prevent excessive server load
			end
		end
	end
})

Tab:AddLabel("Label")

Tab:AddButton({
	Name = "Button!",
	Callback = function()
      		setclipboard(a)
		CoolLabel:Set(a)
  	end    
})
