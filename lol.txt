local Tool
local a

local Window = OrionLib:MakeWindow({Name = "Blox Fruit", HidePremium = false, SaveConfig = true, ConfigFolder = "OrionTest"})

local Tab = Window:MakeTab({
	Name = "Farm",
	Icon = "rbxassetid://4483345998",
	PremiumOnly = false
})

local Section = Tab:AddSection({
	Name = "Setting"
})

Tab:AddDropdown({
	Name = "Tool",
	Default = "Melee",
	Options = {"Melee", "Sword"},
	Callback = function(Value)
		Tool = Value
	end    
})

Tab:AddToggle({
	Name = "Auto Buy Legendary Aura",
	Default = false,
	Callback = function(Value)
	a = game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("CommF_"):InvokeServer("ColorsDealer", "1")
	_G.ABLA = Value
		pcall(function()
			if _G.ABLA and a == "Winter Sky" or a == "Pure Red" or a == "Snow White" then
				repeat game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("CommF_"):InvokeServer("ColorsDealer", "2") wait() until _G.ABLA == false
			end
		end)
	end
})

local CoolLabel = Tab:AddLabel("Label")

Tab:AddButton({
	Name = "Button!",
	Callback = function()
      		setclipboard(a)
		CoolLabel:Set(a)
  	end    
})
